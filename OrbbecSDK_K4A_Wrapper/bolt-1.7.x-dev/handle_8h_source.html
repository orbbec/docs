40/* K4A_DECLARE_CONTEXT creates type matched C functions to create, destroy and get the context. The create and destroy
41functions will ensure matched CPP constructor and destructor are called. To protext against the create function being
42used with CPP and destroy being used with C, or vise-vesa, the types get c or cpp appended to them. */
43#define K4A_DECLARE_CONTEXT(_public_handle_name_, _internal_context_type_) \
44 extern char PRIV_HANDLE_TYPE(_public_handle_name_)[]; \
45 KSELECTANY char PRIV_HANDLE_TYPE(_public_handle_name_)[] = STR_INTERNAL_CONTEXT_TYPE(_internal_context_type_); \
46 typedef struct PUB_HANDLE_TYPE(_public_handle_name_) \
47 { \
48 char *handleType; \
49 _internal_context_type_ context; \
50 } PUB_HANDLE_TYPE(_public_handle_name_); \
51 \
52 /* Define "context_t* handle_t_create(handle_t* handle)" function */ \
53 static inline _internal_context_type_ *_public_handle_name_##_create(_public_handle_name_ *handle) \
54 { \
55 PUB_HANDLE_TYPE(_public_handle_name_) * pContextWrapper; \
56 *handle = NULL; \
57 pContextWrapper = ALLOCATE(PUB_HANDLE_TYPE(_public_handle_name_)); \
58 if (pContextWrapper == NULL) \
59 { \
60 IF_LOGGER(LOG_ERROR("Failed to allocate " #_public_handle_name_, 0);) return NULL; \
61 } \
62 else \
63 { \
64 IF_LOGGER(LOG_TRACE("Created " #_public_handle_name_ " %p", pContextWrapper);) \
65 } \
66 pContextWrapper->handleType = PRIV_HANDLE_TYPE(_public_handle_name_); \
67 *handle = (_public_handle_name_)pContextWrapper; \
68 return &pContextWrapper->context; \
69 } \
70 \
71 /* Define "context_t* handle_t_get_context(handle_t handle)" function */ \
72 static inline _internal_context_type_ *_public_handle_name_##_get_context(_public_handle_name_ handle) \
73 { \
74 if ((handle == NULL) || \
75 ((PUB_HANDLE_TYPE(_public_handle_name_) *)handle)->handleType != PRIV_HANDLE_TYPE(_public_handle_name_)) \
76 { \
77 IF_LOGGER(LOG_ERROR("Invalid " #_public_handle_name_ " %p", handle);) \
78 return NULL; \
79 } \
80 return &(((PUB_HANDLE_TYPE(_public_handle_name_) *)handle)->context); \
81 } \
82 \
83 /* Define "void handle_t_destroy(handle_t handle) function */ \
84 static inline void _public_handle_name_##_destroy(_public_handle_name_ handle) \
85 { \
86 (void)_public_handle_name_##_get_context(handle); \
87 IF_LOGGER(LOG_TRACE("Destroyed " #_public_handle_name_ " %p", handle);) \
88 ((PUB_HANDLE_TYPE(_public_handle_name_) *)handle)->handleType = NULL; \
89 DESTROY((PUB_HANDLE_TYPE(_public_handle_name_) *)handle); \
90 }
