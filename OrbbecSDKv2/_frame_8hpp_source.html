1286// Define the is() template function for the Frame class
1287template <typename T> bool Frame::is() const {
1288 switch(this->getType()) {
1289 case OB_FRAME_IR_LEFT: // Follow
1290 case OB_FRAME_IR_RIGHT: // Follow
1291 case OB_FRAME_IR:
1292 return (typeid(T) == typeid(IRFrame) || typeid(T) == typeid(VideoFrame));
1293 case OB_FRAME_DEPTH:
1294 return (typeid(T) == typeid(DepthFrame) || typeid(T) == typeid(VideoFrame));
1295 case OB_FRAME_COLOR:
1296 case OB_FRAME_COLOR_LEFT:
1297 case OB_FRAME_COLOR_RIGHT:
1298 return (typeid(T) == typeid(ColorFrame) || typeid(T) == typeid(VideoFrame));
1299 case OB_FRAME_CONFIDENCE:
1300 return (typeid(T) == typeid(ConfidenceFrame) || typeid(T) == typeid(VideoFrame));
1301 case OB_FRAME_GYRO:
1302 return (typeid(T) == typeid(GyroFrame));
1303 case OB_FRAME_ACCEL:
1304 return (typeid(T) == typeid(AccelFrame));
1305 case OB_FRAME_POINTS:
1306 return (typeid(T) == typeid(PointsFrame));
1307 case OB_FRAME_LIDAR_POINTS:
1308 return (typeid(T) == typeid(LiDARPointsFrame));
1309 case OB_FRAME_SET:
1310 return (typeid(T) == typeid(FrameSet));
1311 default:
1312 std::cout << "ob::Frame::is() did not catch frame type: " << (int)this->getType() << std::endl;
1313 break;
1314 }
1315 return false;
1316}
