996// Define the is() template function for the Frame class
997template <typename T> bool Frame::is() const {
998 switch(this->getType()) {
999 case OB_FRAME_IR_LEFT: // Follow
1000 case OB_FRAME_IR_RIGHT: // Follow
1001 case OB_FRAME_IR:
1002 return (typeid(T) == typeid(IRFrame) || typeid(T) == typeid(VideoFrame));
1003 case OB_FRAME_DEPTH:
1004 return (typeid(T) == typeid(DepthFrame) || typeid(T) == typeid(VideoFrame));
1005 case OB_FRAME_COLOR:
1006 return (typeid(T) == typeid(ColorFrame) || typeid(T) == typeid(VideoFrame));
1007 case OB_FRAME_GYRO:
1008 return (typeid(T) == typeid(GyroFrame));
1009 case OB_FRAME_ACCEL:
1010 return (typeid(T) == typeid(AccelFrame));
1011 case OB_FRAME_POINTS:
1012 return (typeid(T) == typeid(PointsFrame));
1013 case OB_FRAME_SET:
1014 return (typeid(T) == typeid(FrameSet));
1015 default:
1016 std::cout << "ob::Frame::is() did not catch frame type: " << (int)this->getType() << std::endl;
1017 break;
1018 }
1019 return false;
