1028// Define the is() template function for the Frame class
1029template <typename T> bool Frame::is() const {
1030 switch(this->getType()) {
1031 case OB_FRAME_IR_LEFT: // Follow
1032 case OB_FRAME_IR_RIGHT: // Follow
1033 case OB_FRAME_IR:
1034 return (typeid(T) == typeid(IRFrame) || typeid(T) == typeid(VideoFrame));
1035 case OB_FRAME_DEPTH:
1036 return (typeid(T) == typeid(DepthFrame) || typeid(T) == typeid(VideoFrame));
1037 case OB_FRAME_COLOR:
1038 return (typeid(T) == typeid(ColorFrame) || typeid(T) == typeid(VideoFrame));
1039 case OB_FRAME_GYRO:
1040 return (typeid(T) == typeid(GyroFrame));
1041 case OB_FRAME_ACCEL:
1042 return (typeid(T) == typeid(AccelFrame));
1043 case OB_FRAME_POINTS:
1044 return (typeid(T) == typeid(PointsFrame));
1045 case OB_FRAME_SET:
1046 return (typeid(T) == typeid(FrameSet));
1047 default:
1048 std::cout << "ob::Frame::is() did not catch frame type: " << (int)this->getType() << std::endl;
1049 break;
1050 }
1051 return false;
1052}
