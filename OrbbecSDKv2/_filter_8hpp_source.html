57template <typename T> T getPropertyRange(const OBFilterConfigSchemaItem &item, const double cur) {
58 // If T type is illegal, T will be void
59 using valueType = typename RangeTraits<T>::valueType;
60 T range{};
61 // Compilate error will be reported here if T is void
62 range.cur = static_cast<valueType>(cur);
63 range.def = static_cast<valueType>(item.def);
64 range.max = static_cast<valueType>(item.max);
65 range.min = static_cast<valueType>(item.min);
66 range.step = static_cast<valueType>(item.step);
67 return range;
68}

1000template <typename T> bool Filter::is() {
1001 std::string name = type();
1002 auto it = obFilterTypeMap.find(name);
1003 if(it != obFilterTypeMap.end()) {
1004 return std::type_index(typeid(T)) == it->second;
1005 }
