331template <typename T> bool StreamProfile::is() const {
332 switch(this->getType()) {
333 case OB_STREAM_VIDEO:
334 case OB_STREAM_IR:
335 case OB_STREAM_IR_LEFT:
336 case OB_STREAM_IR_RIGHT:
337 case OB_STREAM_COLOR:
338 case OB_STREAM_DEPTH:
339 case OB_STREAM_RAW_PHASE:
340 return typeid(T) == typeid(VideoStreamProfile);
341 case OB_STREAM_ACCEL:
342 return typeid(T) == typeid(AccelStreamProfile);
343 case OB_STREAM_GYRO:
344 return typeid(T) == typeid(GyroStreamProfile);
345 default:
346 break;
347 }
348 return false;
349}
